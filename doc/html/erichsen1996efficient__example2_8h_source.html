472  void
473  Example2::assemble_on_one_cell(
474  const typename DoFHandler<2, 3>::active_cell_iterator &cell_iter,
475  CellWiseScratchData & scratch,
476  CellWisePerTaskData & data)
477  {
483  data.local_matrix = 0.;
484  // N.B. The construction of the object <code>scratch.fe_values</code> is
485  // carried out in the constructor of <code>CellWiseScratchData</code>
486  scratch.fe_values.reinit(cell_iter);
487 
488  const unsigned int n_q_points = scratch.fe_values.get_quadrature().size();
489  const unsigned int dofs_per_cell = fe.dofs_per_cell;
490 
491  for (unsigned int q = 0; q < n_q_points; q++)
492  {
493  // Iterate over test function DoFs.
494  for (unsigned int i = 0; i < dofs_per_cell; i++)
495  {
496  // Iterate over ansatz function DoFs.
497  for (unsigned int j = 0; j < dofs_per_cell; j++)
498  {
499  data.local_matrix(i, j) +=
500  0.5 * scratch.fe_values.shape_value(i, q) *
501  scratch.fe_values.shape_value(j, q) *
502  scratch.fe_values.JxW(q);
503  }
504  }
505  }
506 
507  // Extract the DoF indices. N.B. Before calling
508  // <code>get_dof_indices</code>, the memory for the argument vector should
509  // have been allocated. Here, the memory for
510  // <code>data.local_dof_indices</code> has been allocated in the
511  // constructor of <code>CellWisePerTaskData</code>.
512  cell_iter->get_dof_indices(data.local_dof_indices);
513  }

534  void
535  Example2::assemble_on_one_pair_of_cells(
536  const typename DoFHandler<2, 3>::active_cell_iterator &kx_cell_iter,
537  const typename DoFHandler<2, 3>::active_cell_iterator &ky_cell_iter,
538  const BEMValues<2, 3> & bem_values,
539  PairCellWiseScratchData & scratch,
540  PairCellWisePerTaskData & data)
541  {
542  // Geometry information.
543  const unsigned int vertices_per_cell = GeometryInfo<2>::vertices_per_cell;
544 
545  // Determine the cell neighboring type based on the vertex dof indices.
546  // The common dof indices will be stored into the vector
547  // <code>vertex_dof_index_intersection</code> if there is any.
548  std::array<types::global_dof_index, vertices_per_cell>
549  kx_vertex_dof_indices(get_vertex_dof_indices<2, 3>(kx_cell_iter));
550  std::array<types::global_dof_index, vertices_per_cell>
551  ky_vertex_dof_indices(get_vertex_dof_indices<2, 3>(ky_cell_iter));
552 
553  scratch.vertex_dof_index_intersection.clear();
554  CellNeighboringType cell_neighboring_type =
555  detect_cell_neighboring_type<2>(kx_vertex_dof_indices,
556  ky_vertex_dof_indices,
557  scratch.vertex_dof_index_intersection);
558 
559  const FiniteElement<2, 3> &kx_fe = kx_cell_iter->get_fe();
560  const FiniteElement<2, 3> &ky_fe = ky_cell_iter->get_fe();
561 
562  const unsigned int kx_n_dofs = kx_fe.dofs_per_cell;
563  const unsigned int ky_n_dofs = ky_fe.dofs_per_cell;
564 
565  // Support points of \f$K_x\f$ and \f$K_y\f$ in the default
566  // hierarchical order.
567  get_hierarchic_support_points_in_real_cell(
568  kx_cell_iter,
569  kx_fe,
570  this->mapping,
571  scratch.kx_support_points_hierarchical);
572  get_hierarchic_support_points_in_real_cell(
573  ky_cell_iter,
574  ky_fe,
575  this->mapping,
576  scratch.ky_support_points_hierarchical);
577 
578  // N.B. The vector holding local DoF indices has to have the right size
579  // before being passed to the function <code>get_dof_indices</code>.
580  kx_cell_iter->get_dof_indices(scratch.kx_local_dof_indices_hierarchical);
581  ky_cell_iter->get_dof_indices(scratch.ky_local_dof_indices_hierarchical);
582 
583  try
584  {
585  // Quadrature rule to be adopted depending on the cell neighboring
586  // type.
587  const QGauss<4> *active_quad_rule = nullptr;
588 
589  switch (cell_neighboring_type)
590  {
591  case SamePanel:
592  {
593  Assert(scratch.vertex_dof_index_intersection.size() ==
594  vertices_per_cell,
595  ExcInternalError());
596 
600  permute_vector(scratch.kx_support_points_hierarchical,
601  scratch.kx_fe_poly_space_numbering_inverse,
602  scratch.kx_support_points_permuted);
603  permute_vector(scratch.ky_support_points_hierarchical,
604  scratch.ky_fe_poly_space_numbering_inverse,
605  scratch.ky_support_points_permuted);
606 
610  permute_vector(scratch.kx_local_dof_indices_hierarchical,
611  scratch.kx_fe_poly_space_numbering_inverse,
612  data.kx_local_dof_indices_permuted);
613  permute_vector(scratch.ky_local_dof_indices_hierarchical,
614  scratch.ky_fe_poly_space_numbering_inverse,
615  data.ky_local_dof_indices_permuted);
616 
620  active_quad_rule = &(bem_values.quad_rule_for_same_panel);
621 
626  for (unsigned int k3_index = 0; k3_index < 8; k3_index++)
627  {
628  for (unsigned int q = 0; q < active_quad_rule->size();
629  q++)
630  {
631  scratch.kx_jacobians_same_panel(k3_index, q) =
632  surface_jacobian_det_and_normal_vector(
633  k3_index,
634  q,
635  bem_values
636  .kx_shape_grad_matrix_table_for_same_panel,
637  scratch.kx_support_points_permuted,
638  scratch.kx_normals_same_panel(k3_index, q));
639 
640  scratch.ky_jacobians_same_panel(k3_index, q) =
641  surface_jacobian_det_and_normal_vector(
642  k3_index,
643  q,
644  bem_values
645  .ky_shape_grad_matrix_table_for_same_panel,
646  scratch.ky_support_points_permuted,
647  scratch.ky_normals_same_panel(k3_index, q));
648 
649  scratch.kx_quad_points_same_panel(k3_index, q) =
650  transform_unit_to_permuted_real_cell(
651  k3_index,
652  q,
653  bem_values.kx_shape_value_table_for_same_panel,
654  scratch.kx_support_points_permuted);
655 
656  scratch.ky_quad_points_same_panel(k3_index, q) =
657  transform_unit_to_permuted_real_cell(
658  k3_index,
659  q,
660  bem_values.ky_shape_value_table_for_same_panel,
661  scratch.ky_support_points_permuted);
662  }
663  }
664 
665  break;
666  }
667  case CommonEdge:
668  {
669  // This part handles the common edge case of Sauter's
670  // quadrature rule.
671  // 1. Get the DoF indices in the lexicographic order for \f$K_x\f$.
672  // 2. Get the DoF indices in the reversed lexicographic order
673  // for \f$K_y\f$.
674  // 3. Extract DoF indices only for cell vertices in \f$K_x\f$ and
675  // \f$K_y\f$. N.B. The DoF indices for the last two vertices are
676  // swapped, such that the four vertices are in clockwise or
677  // counter clockwise order.
678  // 4. Determine the starting vertex.
679 
680  Assert(scratch.vertex_dof_index_intersection.size() ==
681  GeometryInfo<2>::vertices_per_face,
682  ExcInternalError());
683 
684  permute_vector(scratch.kx_local_dof_indices_hierarchical,
685  scratch.kx_fe_poly_space_numbering_inverse,
686  data.kx_local_dof_indices_permuted);
687 
688  permute_vector(
689  scratch.ky_local_dof_indices_hierarchical,
690  scratch.ky_fe_reversed_poly_space_numbering_inverse,
691  data.ky_local_dof_indices_permuted);
692 
693  std::array<types::global_dof_index, vertices_per_cell>
694  kx_local_vertex_dof_indices_swapped;
695  get_vertex_dof_indices_swapped<2, 3>(
696  kx_fe,
697  data.kx_local_dof_indices_permuted,
698  kx_local_vertex_dof_indices_swapped);
699  std::array<types::global_dof_index, vertices_per_cell>
700  ky_local_vertex_dof_indices_swapped;
701  get_vertex_dof_indices_swapped<2, 3>(
702  ky_fe,
703  data.ky_local_dof_indices_permuted,
704  ky_local_vertex_dof_indices_swapped);
705 
706  // Determine the starting vertex index in \f$K_x\f$ and \f$K_y\f$.
707  unsigned int kx_starting_vertex_index =
708  get_start_vertex_dof_index<vertices_per_cell>(
709  scratch.vertex_dof_index_intersection,
710  kx_local_vertex_dof_indices_swapped);
711  Assert(kx_starting_vertex_index < vertices_per_cell,
712  ExcInternalError());
713  unsigned int ky_starting_vertex_index =
714  get_start_vertex_dof_index<vertices_per_cell>(
715  scratch.vertex_dof_index_intersection,
716  ky_local_vertex_dof_indices_swapped);
717  Assert(ky_starting_vertex_index < vertices_per_cell,
718  ExcInternalError());
719 
720  // Generate the permutation of DoFs in \f$K_x\f$ and \f$K_y\f$ by
721  // starting from <code>kx_starting_vertex_index</code> or
722  // <code>ky_starting_vertex_index</code>.
723  generate_forward_dof_permutation(
724  kx_fe,
725  kx_starting_vertex_index,
726  scratch.kx_local_dof_permutation);
727  generate_backward_dof_permutation(
728  ky_fe,
729  ky_starting_vertex_index,
730  scratch.ky_local_dof_permutation);
731 
732  permute_vector(scratch.kx_support_points_hierarchical,
