316  Example2::calc_cell_neighboring_types()
317  {
318  const unsigned int n_active_cells = triangulation.n_active_cells();
319  FullMatrix<unsigned int> cell_neighboring_type_matrix(n_active_cells,
320  n_active_cells);
321 
322  types::global_vertex_index i = 0;
323  for (const auto &first_cell : triangulation.active_cell_iterators())
324  {
325  std::array<types::global_vertex_index,
326  GeometryInfo<2>::vertices_per_cell>
327  first_cell_vertex_indices(get_vertex_indices<2, 3>(first_cell));
328 
329  types::global_vertex_index j = 0;
330  for (const auto &second_cell : triangulation.active_cell_iterators())
331  {
332  std::array<types::global_vertex_index,
333  GeometryInfo<2>::vertices_per_cell>
334  second_cell_vertex_indices(
335  get_vertex_indices<2, 3>(second_cell));
336 
337  std::vector<types::global_vertex_index> vertex_index_intersection;
338  vertex_index_intersection.reserve(
339  GeometryInfo<2>::vertices_per_cell);
340  cell_neighboring_type_matrix(i, j) =
341  LaplaceBEM::detect_cell_neighboring_type<2>(
342  first_cell_vertex_indices,
343  second_cell_vertex_indices,
344  vertex_index_intersection);
345 
346  j++;
347  }
348 
349  i++;
350  }
351 
352  deallog << "Cell neighboring types..." << std::endl;
353  cell_neighboring_type_matrix.print(deallog);
354  deallog << std::endl;

385  void
386  Example2::assemble_on_one_cell(
387  const typename DoFHandler<2, 3>::active_cell_iterator &cell_iter,
388  CellWiseScratchData & scratch,
389  CellWisePerTaskData & data)
390  {
391  // Clear the local matrix in case that it is reused from another finished
392  // task. N.B. Its memory has already been allocated in the constructor of
393  // <code>CellWisePerTaskData</code>.
394  data.local_matrix = 0.;
395  // N.B. The construction of the object <code>scratch.fe_values</code> is
396  // carried out in the constructor of <code>CellWiseScratchData</code>
397  scratch.fe_values.reinit(cell_iter);
398 
399  const unsigned int n_q_points = scratch.fe_values.get_quadrature().size();
400  const unsigned int dofs_per_cell = fe.dofs_per_cell;
401 
402  for (unsigned int q = 0; q < n_q_points; q++)
403  {
404  // Iterate over test function DoFs.
405  for (unsigned int i = 0; i < dofs_per_cell; i++)
406  {
407  // Iterate over ansatz function DoFs.
408  for (unsigned int j = 0; j < dofs_per_cell; j++)
409  {
410  data.local_matrix(i, j) +=
411  0.5 * scratch.fe_values.shape_value(i, q) *
412  scratch.fe_values.shape_value(j, q) *
413  scratch.fe_values.JxW(q);
414  }
415  }
416  }
417 
418  // Extract the DoF indices. N.B. Before calling
419  // <code>get_dof_indices</code>, the memory for the argument vector should
420  // have been allocated. Here, the memory for
421  // <code>data.local_dof_indices</code> has been allocated in the
422  // constructor of <code>CellWisePerTaskData</code>.
423  cell_iter->get_dof_indices(data.local_dof_indices);
424  }

445  void
446  Example2::assemble_on_one_pair_of_cells(
447  const typename DoFHandler<2, 3>::active_cell_iterator &kx_cell_iter,
448  const typename DoFHandler<2, 3>::active_cell_iterator &ky_cell_iter,
449  const BEMValues<2, 3> & bem_values,
450  PairCellWiseScratchData & scratch,
451  PairCellWisePerTaskData & data)
452  {
453  // Geometry information.
454  const unsigned int vertices_per_cell = GeometryInfo<2>::vertices_per_cell;
455 
456  // Determine the cell neighboring type based on the vertex dof indices.
457  // The common dof indices will be stored into the vector
458  // <code>vertex_dof_index_intersection</code> if there is any.
459  std::array<types::global_dof_index, vertices_per_cell>
460  kx_vertex_dof_indices(get_vertex_dof_indices<2, 3>(kx_cell_iter));
461  std::array<types::global_dof_index, vertices_per_cell>
462  ky_vertex_dof_indices(get_vertex_dof_indices<2, 3>(ky_cell_iter));
463 
464  scratch.vertex_dof_index_intersection.clear();
465  CellNeighboringType cell_neighboring_type =
466  detect_cell_neighboring_type<2>(kx_vertex_dof_indices,
467  ky_vertex_dof_indices,
468  scratch.vertex_dof_index_intersection);
469 
470  const FiniteElement<2, 3> &kx_fe = kx_cell_iter->get_fe();
471  const FiniteElement<2, 3> &ky_fe = ky_cell_iter->get_fe();
472 
473  const unsigned int kx_n_dofs = kx_fe.dofs_per_cell;
474  const unsigned int ky_n_dofs = ky_fe.dofs_per_cell;
475 
476  // Support points of $K_x$ and $K_y$ in the default
477  // hierarchical order.
478  hierarchical_support_points_in_real_cell(
479  kx_cell_iter,
480  kx_fe,
481  this->mapping,
482  scratch.kx_support_points_hierarchical);
483  hierarchical_support_points_in_real_cell(
484  ky_cell_iter,
485  ky_fe,
486  this->mapping,
487  scratch.ky_support_points_hierarchical);
488 
489  // N.B. The vector holding local DoF indices has to have the right size
490  // before being passed to the function <code>get_dof_indices</code>.
491  kx_cell_iter->get_dof_indices(scratch.kx_local_dof_indices_hierarchical);
492  ky_cell_iter->get_dof_indices(scratch.ky_local_dof_indices_hierarchical);
493 
494  try
495  {
496  // Quadrature rule to be adopted depending on the cell neighboring
497  // type.
498  const QGauss<4> *active_quad_rule = nullptr;
499 
500  switch (cell_neighboring_type)
501  {
502  case SamePanel:
503  {
504  Assert(scratch.vertex_dof_index_intersection.size() ==
505  vertices_per_cell,
506  ExcInternalError());
507 
508  // Get support points in tensor product oder.
509  permute_vector(scratch.kx_support_points_hierarchical,
510  scratch.kx_fe_poly_space_numbering_inverse,
511  scratch.kx_support_points_permuted);
512  permute_vector(scratch.ky_support_points_hierarchical,
513  scratch.ky_fe_poly_space_numbering_inverse,
514  scratch.ky_support_points_permuted);
515 
516  // Get permuted local DoF indices.
517  permute_vector(scratch.kx_local_dof_indices_hierarchical,
518  scratch.kx_fe_poly_space_numbering_inverse,
519  data.kx_local_dof_indices_permuted);
520  permute_vector(scratch.ky_local_dof_indices_hierarchical,
521  scratch.ky_fe_poly_space_numbering_inverse,
522  data.ky_local_dof_indices_permuted);
523 
524  active_quad_rule = &(bem_values.quad_rule_for_same_panel);
525 
526  // Precalculate surface Jacobians and normal vectors.
527  for (unsigned int k3_index = 0; k3_index < 8; k3_index++)
528  {
529  for (unsigned int q = 0; q < active_quad_rule->size();
530  q++)
531  {
532  scratch.kx_jacobians_same_panel(k3_index, q) =
533  surface_jacobian_det_and_normal_vector(
534  k3_index,
535  q,
536  bem_values
537  .kx_shape_grad_matrix_table_for_same_panel,
538  scratch.kx_support_points_permuted,
539  scratch.kx_normals_same_panel(k3_index, q));
540 
541  scratch.ky_jacobians_same_panel(k3_index, q) =
542  surface_jacobian_det_and_normal_vector(
543  k3_index,
544  q,
545  bem_values
546  .ky_shape_grad_matrix_table_for_same_panel,
547  scratch.ky_support_points_permuted,
548  scratch.ky_normals_same_panel(k3_index, q));
549 
550  scratch.kx_quad_points_same_panel(k3_index, q) =
551  transform_unit_to_permuted_real_cell(
552  k3_index,
553  q,
554  bem_values.kx_shape_value_table_for_same_panel,
555  scratch.kx_support_points_permuted);
556 
557  scratch.ky_quad_points_same_panel(k3_index, q) =
558  transform_unit_to_permuted_real_cell(
559  k3_index,
560  q,
561  bem_values.ky_shape_value_table_for_same_panel,
562  scratch.ky_support_points_permuted);
563  }
564  }
565 
566  break;
567  }
568  case CommonEdge:
569  {
570  // This part handles the common edge case of Sauter's
571  // quadrature rule.
572  // 1. Get the DoF indices in tensor product order for $K_x$.
573  // 2. Get the DoF indices in reversed tensor product order for
574  // $K_x$.
575  // 3. Extract DoF indices only for cell vertices in $K_x$ and
576  // $K_y$. N.B. The DoF indices for the last two vertices are
577  // swapped, such that the four vertices are in clockwise or
578  // counter clockwise order.
579  // 4. Determine the starting vertex.
580 
581  Assert(scratch.vertex_dof_index_intersection.size() ==
582  GeometryInfo<2>::vertices_per_face,
583  ExcInternalError());
584 
585  permute_vector(scratch.kx_local_dof_indices_hierarchical,
586  scratch.kx_fe_poly_space_numbering_inverse,
587  data.kx_local_dof_indices_permuted);
588 
589  permute_vector(
590  scratch.ky_local_dof_indices_hierarchical,
591  scratch.ky_fe_reversed_poly_space_numbering_inverse,
592  data.ky_local_dof_indices_permuted);
593 
594  std::array<types::global_dof_index, vertices_per_cell>
595  kx_local_vertex_dof_indices_swapped;
596  get_vertex_dof_indices_swapped<2, 3>(
597  kx_fe,
598  data.kx_local_dof_indices_permuted,
599  kx_local_vertex_dof_indices_swapped);
600  std::array<types::global_dof_index, vertices_per_cell>
601  ky_local_vertex_dof_indices_swapped;
602  get_vertex_dof_indices_swapped<2, 3>(
603  ky_fe,
604  data.ky_local_dof_indices_permuted,
605  ky_local_vertex_dof_indices_swapped);
606 
607  // Determine the starting vertex index in $K_x$ and $K_y$.
608  unsigned int kx_starting_vertex_index =
609  get_start_vertex_dof_index<vertices_per_cell>(
610  scratch.vertex_dof_index_intersection,
611  kx_local_vertex_dof_indices_swapped);
612  Assert(kx_starting_vertex_index < vertices_per_cell,
613  ExcInternalError());
614  unsigned int ky_starting_vertex_index =
615  get_start_vertex_dof_index<vertices_per_cell>(
616  scratch.vertex_dof_index_intersection,
617  ky_local_vertex_dof_indices_swapped);
618  Assert(ky_starting_vertex_index < vertices_per_cell,
619  ExcInternalError());
620 
621  // Generate the permutation of DoFs in $K_x$ and $K_y$ by
622  // starting from <code>kx_starting_vertex_index</code> or
623  // <code>ky_starting_vertex_index</code>.
624  generate_forward_dof_permutation(
625  kx_fe,
626  kx_starting_vertex_index,
627  scratch.kx_local_dof_permutation);
628  generate_backward_dof_permutation(
629  ky_fe,
630  ky_starting_vertex_index,
631  scratch.ky_local_dof_permutation);
632 
633  permute_vector(scratch.kx_support_points_hierarchical,
634  scratch.kx_local_dof_permutation,
635  scratch.kx_support_points_permuted);
636  permute_vector(scratch.ky_support_points_hierarchical,
637  scratch.ky_local_dof_permutation,
638  scratch.ky_support_points_permuted);
639 
640  permute_vector(scratch.kx_local_dof_indices_hierarchical,
641  scratch.kx_local_dof_permutation,
642  data.kx_local_dof_indices_permuted);
643  permute_vector(scratch.ky_local_dof_indices_hierarchical,
644  scratch.ky_local_dof_permutation,
645  data.ky_local_dof_indices_permuted);
646 
647  active_quad_rule = &(bem_values.quad_rule_for_common_edge);
648 
649  // Precalculate surface Jacobians and normal vectors.
650  for (unsigned int k3_index = 0; k3_index < 6; k3_index++)
651  {
652  for (unsigned int q = 0; q < active_quad_rule->size();
653  q++)
654  {
655  scratch.kx_jacobians_common_edge(k3_index, q) =
656  surface_jacobian_det_and_normal_vector(
657  k3_index,
658  q,
659  bem_values
660  .kx_shape_grad_matrix_table_for_common_edge,
661  scratch.kx_support_points_permuted,
662  scratch.kx_normals_common_edge(k3_index, q));
663 
664  scratch.ky_jacobians_common_edge(k3_index, q) =
665  surface_jacobian_det_and_normal_vector(
666  k3_index,
667  q,
668  bem_values
669  .ky_shape_grad_matrix_table_for_common_edge,
670  scratch.ky_support_points_permuted,
671  scratch.ky_normals_common_edge(k3_index, q));
672 
673  scratch.kx_quad_points_common_edge(k3_index, q) =
674  transform_unit_to_permuted_real_cell(
675  k3_index,
676  q,
677  bem_values.kx_shape_value_table_for_common_edge,
678  scratch.kx_support_points_permuted);
679 
680  scratch.ky_quad_points_common_edge(k3_index, q) =
681  transform_unit_to_permuted_real_cell(
682  k3_index,
683  q,
684  bem_values.ky_shape_value_table_for_common_edge,
685  scratch.ky_support_points_permuted);
686  }
687  }
688 
689  break;
690  }
691  case CommonVertex:
692  {
693  Assert(scratch.vertex_dof_index_intersection.size() == 1,
694  ExcInternalError());
695 
696  permute_vector(scratch.kx_local_dof_indices_hierarchical,
697  scratch.kx_fe_poly_space_numbering_inverse,
698  data.kx_local_dof_indices_permuted);
699 
700  permute_vector(scratch.ky_local_dof_indices_hierarchical,
701  scratch.ky_fe_poly_space_numbering_inverse,
702  data.ky_local_dof_indices_permuted);
703 
704  std::array<types::global_dof_index, vertices_per_cell>
705  kx_local_vertex_dof_indices_swapped;
706  get_vertex_dof_indices_swapped<2, 3>(
707  kx_fe,
708  data.kx_local_dof_indices_permuted,
709  kx_local_vertex_dof_indices_swapped);
710  std::array<types::global_dof_index, vertices_per_cell>
711  ky_local_vertex_dof_indices_swapped;
712  get_vertex_dof_indices_swapped<2, 3>(
713  ky_fe,
714  data.ky_local_dof_indices_permuted,
715  ky_local_vertex_dof_indices_swapped);
716 
717  // Determine the starting vertex index in $K_x$ and $K_y$.
718  unsigned int kx_starting_vertex_index =
719  get_start_vertex_dof_index<vertices_per_cell>(
720  scratch.vertex_dof_index_intersection,
721  kx_local_vertex_dof_indices_swapped);
722  Assert(kx_starting_vertex_index < vertices_per_cell,
723  ExcInternalError());
724  unsigned int ky_starting_vertex_index =
725  get_start_vertex_dof_index<vertices_per_cell>(
726  scratch.vertex_dof_index_intersection,
727  ky_local_vertex_dof_indices_swapped);
728  Assert(ky_starting_vertex_index < vertices_per_cell,
729  ExcInternalError());
730 
731  // Generate the permutation of DoFs in $K_x$ and $K_y$ by
732  // starting from <code>kx_starting_vertex_index</code> or
733  // <code>ky_starting_vertex_index</code>.
734  generate_forward_dof_permutation(
735  kx_fe,
736  kx_starting_vertex_index,
737  scratch.kx_local_dof_permutation);
738  generate_forward_dof_permutation(
739  ky_fe,
740  ky_starting_vertex_index,
741  scratch.ky_local_dof_permutation);
742 
743  permute_vector(scratch.kx_support_points_hierarchical,
744  scratch.kx_local_dof_permutation,
745  scratch.kx_support_points_permuted);
746  permute_vector(scratch.ky_support_points_hierarchical,
747  scratch.ky_local_dof_permutation,
748  scratch.ky_support_points_permuted);
749 
750  permute_vector(scratch.kx_local_dof_indices_hierarchical,
751  scratch.kx_local_dof_permutation,
752  data.kx_local_dof_indices_permuted);
753  permute_vector(scratch.ky_local_dof_indices_hierarchical,
754  scratch.ky_local_dof_permutation,
755  data.ky_local_dof_indices_permuted);
756 
757  active_quad_rule = &(bem_values.quad_rule_for_common_vertex);
758 
759  // Precalculate surface Jacobians and normal vectors.
760  for (unsigned int k3_index = 0; k3_index < 4; k3_index++)
761  {
762  for (unsigned int q = 0; q < active_quad_rule->size();
763  q++)
764  {
765  scratch.kx_jacobians_common_vertex(k3_index, q) =
766  surface_jacobian_det_and_normal_vector(
767  k3_index,
768  q,
769  bem_values
770  .kx_shape_grad_matrix_table_for_common_vertex,
771  scratch.kx_support_points_permuted,
772  scratch.kx_normals_common_vertex(k3_index, q));
773 
774  scratch.ky_jacobians_common_vertex(k3_index, q) =
775  surface_jacobian_det_and_normal_vector(
776  k3_index,
777  q,
778  bem_values
779  .ky_shape_grad_matrix_table_for_common_vertex,
780  scratch.ky_support_points_permuted,
781  scratch.ky_normals_common_vertex(k3_index, q));
782 
783  scratch.kx_quad_points_common_vertex(k3_index, q) =
784  transform_unit_to_permuted_real_cell(
785  k3_index,
786  q,
787  bem_values.kx_shape_value_table_for_common_vertex,
788  scratch.kx_support_points_permuted);
789 
790  scratch.ky_quad_points_common_vertex(k3_index, q) =
791  transform_unit_to_permuted_real_cell(
792  k3_index,
793  q,
794  bem_values.ky_shape_value_table_for_common_vertex,
795  scratch.ky_support_points_permuted);
796  }
797  }
798 
799  break;
800  }
801  case Regular:
802  {
803  Assert(scratch.vertex_dof_index_intersection.size() == 0,
804  ExcInternalError());
805 
806  permute_vector(scratch.kx_local_dof_indices_hierarchical,
807  scratch.kx_fe_poly_space_numbering_inverse,
808  data.kx_local_dof_indices_permuted);
809 
810  permute_vector(scratch.ky_local_dof_indices_hierarchical,
811  scratch.ky_fe_poly_space_numbering_inverse,
812  data.ky_local_dof_indices_permuted);
813 
814  permute_vector(scratch.kx_support_points_hierarchical,
815  scratch.kx_fe_poly_space_numbering_inverse,
816  scratch.kx_support_points_permuted);
817  permute_vector(scratch.ky_support_points_hierarchical,
818  scratch.ky_fe_poly_space_numbering_inverse,
819  scratch.ky_support_points_permuted);
820 
821  active_quad_rule = &(bem_values.quad_rule_for_regular);
822 
823  // Precalculate surface Jacobians and normal vectors.
824  for (unsigned int q = 0; q < active_quad_rule->size(); q++)
825  {
826  scratch.kx_jacobians_regular(0, q) =
827  surface_jacobian_det_and_normal_vector(
828  0,
829  q,
830  bem_values.kx_shape_grad_matrix_table_for_regular,
831  scratch.kx_support_points_permuted,
832  scratch.kx_normals_regular(0, q));
833 
834  scratch.ky_jacobians_regular(0, q) =
835  surface_jacobian_det_and_normal_vector(
836  0,
837  q,
838  bem_values.ky_shape_grad_matrix_table_for_regular,
839  scratch.ky_support_points_permuted,
840  scratch.ky_normals_regular(0, q));
841 
842  scratch.kx_quad_points_regular(0, q) =
843  transform_unit_to_permuted_real_cell(
844  0,
845  q,
846  bem_values.kx_shape_value_table_for_regular,
847  scratch.kx_support_points_permuted);
848 
849  scratch.ky_quad_points_regular(0, q) =
850  transform_unit_to_permuted_real_cell(
851  0,
852  q,
853  bem_values.ky_shape_value_table_for_regular,
854  scratch.ky_support_points_permuted);
855  }
856 
857  break;
858  }
859  default:
860  {
861  Assert(false, ExcNotImplemented());
862  active_quad_rule = nullptr;
863  }
864  }
865 
866  // Clear the local matrix in case that it is reused from another
867  // finished task. N.B. Its memory has already been allocated in the
868  // constructor of <code>CellPairWisePerTaskData</code>.
869  data.dlp_matrix = 0.;
870  data.slp_matrix = 0.;
871 
872  // Iterate over DoFs for test function space in tensor product
873  // order in $K_x$.
874  for (unsigned int i = 0; i < kx_n_dofs; i++)
875  {
876  // Iterate over DoFs for ansatz function space in tensor
877  // product order in $K_y$.
878  for (unsigned int j = 0; j < ky_n_dofs; j++)
879  {
880  // Pullback the kernel function to unit cell.
881  KernelPulledbackToUnitCell<2, 3, double>
882  dlp_kernel_pullback_on_unit(
883  this->dlp,
884  cell_neighboring_type,
885  scratch.kx_support_points_permuted,
886  scratch.ky_support_points_permuted,
887  kx_fe,
888  ky_fe,
889  &bem_values,
890  &scratch,
891  i,
892  j);
893 
894  // Pullback the kernel function to Sauter parameter
895  // space.
896  KernelPulledbackToSauterSpace<2, 3, double>
897  dlp_kernel_pullback_on_sauter(dlp_kernel_pullback_on_unit,
898  cell_neighboring_type,
899  &bem_values);
900 
901  // Apply 4d Sauter numerical quadrature.
902  data.dlp_matrix(i, j) = ApplyQuadratureUsingBEMValues(
903  *active_quad_rule, dlp_kernel_pullback_on_sauter);
904 
905  // Pullback the kernel function to unit cell.
906  KernelPulledbackToUnitCell<2, 3, double>
907  slp_kernel_pullback_on_unit(
908  this->slp,
909  cell_neighboring_type,
910  scratch.kx_support_points_permuted,
911  scratch.ky_support_points_permuted,
912  kx_fe,
913  ky_fe,
914  &bem_values,
915  &scratch,
916  i,
917  j);
918 
919  // Pullback the kernel function to Sauter parameter
920  // space.
921  KernelPulledbackToSauterSpace<2, 3, double>
922  slp_kernel_pullback_on_sauter(slp_kernel_pullback_on_unit,
923  cell_neighboring_type,
924  &bem_values);
925 
926  // Apply 4d Sauter numerical quadrature.
927  data.slp_matrix(i, j) = ApplyQuadratureUsingBEMValues(
928  *active_quad_rule, slp_kernel_pullback_on_sauter);
929  }
930  }
931  }
932  catch (const std::bad_cast &e)
933  {
934  Assert(false, ExcInternalError());
935  }

943  const unsigned int ky_dofs_per_cell = data.dlp_matrix.n();

961  void
962  Example2::assemble_system_smp()
963  {
964  MultithreadInfo::set_thread_limit(4);
965 
966  // Generate normal Gauss-Legendre quadrature rule for FEM integration.
967  QGauss<2> quadrature_formula_2d(fe.degree + 1);
968 
969 #ifdef GRAPH_COLORING
970  // Graph coloring of the mesh for parallelizing matrix assembly.
971  std::vector<std::vector<typename DoFHandler<2, 3>::active_cell_iterator>>
972  colored_cells = GraphColoring::make_graph_coloring(
973  dof_handler.begin_active(),
974  dof_handler.end(),
975  (std::function<std::vector<types::global_dof_index>(
976  const typename DoFHandler<2, 3>::active_cell_iterator &)>)
977  get_conflict_indices<2, 3>);
978 
979  WorkStream::run(colored_cells,
980  std::bind(&Example2::assemble_on_one_cell,
981  this,
982  std::placeholders::_1,
983  std::placeholders::_2,
984  std::placeholders::_3),
985  std::bind(&Example2::copy_cell_local_to_global,
986  this,
987  std::placeholders::_1),
988  CellWiseScratchData(fe,
989  quadrature_formula_2d,
990  update_values | update_JxW_values),
991  CellWisePerTaskData(fe));
992 #else
993  WorkStream::run(dof_handler.begin_active(),
994  dof_handler.end(),
995  std::bind(&Example2::assemble_on_one_cell,
996  this,
997  std::placeholders::_1,
998  std::placeholders::_2,
999  std::placeholders::_3),
1000  std::bind(&Example2::copy_cell_local_to_global,
1001  this,
1002  std::placeholders::_1),
1003  CellWiseScratchData(fe,
1004  quadrature_formula_2d,
1005  update_values | update_JxW_values),
1006  CellWisePerTaskData(fe));
1007 #endif
1008 
1009 
1010 
1011  // Precalculate shape function values and their gradient
1012  // values at each quadrature point. N.B.
1013  // 1. The data tables for shape function values and their gradient values
1014  // should be calculated for both function space on $K_x$ and function
1015  // space on $K_y$.
1016  // 2. Being different from the integral in FEM, the integral in BEM
1017  // handled by Sauter's quadrature rule has multiple parts of $k_3$ (except
1018  // the regular cell neighboring type), each of which should be evaluated
1019  // at a different set of quadrature points in the unit cell after
1020  // coordinate transformation from the parametric space. Therefore, a
1021  // dimension with respect to $k_3$ term index should be added to the data
1022  // table compared to the usual FEValues.
1023 
1024  // Generate 4D Gauss-Legendre quadrature rules for various cell
1025  // neighboring types.
1026  const unsigned int quad_order_for_same_panel = 5;
1027  const unsigned int quad_order_for_common_edge = 4;
1028  const unsigned int quad_order_for_common_vertex = 4;
1029  const unsigned int quad_order_for_regular = 3;
1030 
1031  QGauss<4> quad_rule_for_same_panel(quad_order_for_same_panel);
1032  QGauss<4> quad_rule_for_common_edge(quad_order_for_common_edge);
1033  QGauss<4> quad_rule_for_common_vertex(quad_order_for_common_vertex);
1034  QGauss<4> quad_rule_for_regular(quad_order_for_regular);
1035 
1036  // Precalculate data tables for shape values at quadrature points.
1037  BEMValues<2, 3> bem_values(fe,
1038  fe,
1039  quad_rule_for_same_panel,
1040  quad_rule_for_common_edge,
1041  quad_rule_for_common_vertex,
1042  quad_rule_for_regular);
1043  bem_values.fill_shape_value_tables();
1044  bem_values.fill_shape_grad_matrix_tables();
1045 
1046  // Initialize the progress display.
1047  boost::progress_display pd(triangulation.n_active_cells(),
1048  deallog.get_console());
1049 
1050  PairCellWiseScratchData scratch_data(fe, fe, bem_values);
1051  PairCellWisePerTaskData per_task_data(fe, fe);
1052  // Calculate the term $(v, Ku)$ to be assembled into system matrix and
1053  // $(v, V(\psi))$ to be assembled into right-hand side vector.
1054  for (const auto &e : dof_handler.active_cell_iterators())
1055  {
1056 #ifdef GRAPH_COLORING
1057  WorkStream::run(
1058  colored_cells,
1059  std::bind(&Example2::assemble_on_one_pair_of_cells,
1060  this,
1061  e,
1062  std::placeholders::_1,
1063  bem_values,
1064  std::placeholders::_2,
1065  std::placeholders::_3),
1066  std::bind(&Example2::copy_pair_of_cells_local_to_global,
1067  this,
1068  std::placeholders::_1),
1069  scratch_data,
1070  per_task_data);
1071 #else
1072  WorkStream::run(
1073  dof_handler.begin_active(),
1074  dof_handler.end(),
1075  std::bind(&Example2::assemble_on_one_pair_of_cells,
1076  this,
1077  e,
1078  std::placeholders::_1,
1079  bem_values,
1080  std::placeholders::_2,
1081  std::placeholders::_3),
1082  std::bind(&Example2::copy_pair_of_cells_local_to_global,
1083  this,
1084  std::placeholders::_1),
1085  scratch_data,
1086  per_task_data);
1087 #endif
1088 
1089  ++pd;
1090  }
1091 
1092  // Calculate the right-hand side vector.
1093  system_rhs_matrix.vmult(system_rhs, neumann_bc);
1094  }

1097  void
1098  Example2::assemble_system_serial()
1099  {
1100  // Generate normal Gauss-Legendre quadrature rule for FEM integration.
1101  QGauss<2> quadrature_formula_2d(fe.degree + 1);
1102 
1103  const unsigned int dofs_per_cell = fe.dofs_per_cell;
1104 
1105  FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
1106 
1107  // Calculate the term $(v, \frac{1}{2}u)$.
1108  FEValues<2, 3> fe_values(fe,
1109  quadrature_formula_2d,
1110  update_values | update_JxW_values);
1111 
1112  // The memory of this vector should be allocated before the calling of
1113  // <code>get_dof_indices</code>.
1114  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
1115 
1116  const unsigned int n_q_points = quadrature_formula_2d.size();
1117 
1118  // Loop over each active cell.
1119  for (const auto &cell : dof_handler.active_cell_iterators())
1120  {
1121  // Clear the local matrix elements for a new cell.
1122  local_matrix = 0.;
1123 
1124  fe_values.reinit(cell);
1125 
1126  // Loop over each quadrature point in the current cell.
1127  for (unsigned int q = 0; q < n_q_points; q++)
1128  {
1129  // Iterate over test function DoFs.
1130  for (unsigned int i = 0; i < dofs_per_cell; i++)
1131  {
1132  // Iterate over ansatz function DoFs.
1133  for (unsigned int j = 0; j < dofs_per_cell; j++)
1134  {
1135  local_matrix(i, j) += 0.5 * fe_values.shape_value(i, q) *
1136  fe_values.shape_value(j, q) *
1137  fe_values.JxW(q);
1138  }
1139  }
1140  }
1141 
1142  // Extract the DoF indices.
1143  cell->get_dof_indices(local_dof_indices);
1144 
1145  // Assemble the local matrix to system matrix.
1146  for (unsigned int i = 0; i < dofs_per_cell; i++)
1147  {
1148  for (unsigned int j = 0; j < dofs_per_cell; j++)
1149  {
1150  system_matrix.add(local_dof_indices[i],
1151  local_dof_indices[j],
1152  local_matrix(i, j));
1153  }
1154  }
1155  }
1156 
1157  // Precalculate shape function values and their gradient
1158  // values at each quadrature point. N.B.
1159  // 1. The data tables for shape function values and their gradient values
1160  // should be calculated for both function space on $K_x$ and function
1161  // space on $K_y$.
1162  // 2. Being different from the integral in FEM, the integral in BEM
1163  // handled by Sauter's quadrature rule has multiple parts of $k_3$ (except
1164  // the regular cell neighboring type), each of which should be evaluated
1165  // at a different set of quadrature points in the unit cell after
1166  // coordinate transformation from the parametric space. Therefore, a
1167  // dimension with respect to $k_3$ term index should be added to the data
1168  // table compared to the usual FEValues.
1169 
1170  // Generate 4D Gauss-Legendre quadrature rules for various cell
1171  // neighboring types.
1172  const unsigned int quad_order_for_same_panel = 5;
1173  const unsigned int quad_order_for_common_edge = 4;
1174  const unsigned int quad_order_for_common_vertex = 4;
1175  const unsigned int quad_order_for_regular = 3;
1176 
1177  QGauss<4> quad_rule_for_same_panel(quad_order_for_same_panel);
1178  QGauss<4> quad_rule_for_common_edge(quad_order_for_common_edge);
1179  QGauss<4> quad_rule_for_common_vertex(quad_order_for_common_vertex);
1180  QGauss<4> quad_rule_for_regular(quad_order_for_regular);
1181 
1182  // Precalculate data tables for shape values at quadrature points.
1183  BEMValues<2, 3> bem_values(fe,
1184  fe,
1185  quad_rule_for_same_panel,
1186  quad_rule_for_common_edge,
1187  quad_rule_for_common_vertex,
1188  quad_rule_for_regular);
1189  bem_values.fill_shape_value_tables();
1190  bem_values.fill_shape_grad_matrix_tables();
1191 
1192  // Initialize the progress display.
1193  boost::progress_display pd(triangulation.n_active_cells() *
1194  triangulation.n_active_cells(),
1195  deallog.get_console());
1196 
1197  // Calculate the term $(v, Ku)$ to be assembled into system matrix and
1198  // $(v, V(\psi))$ to be assembled into right-hand side vector.
1199  for (const auto &e : dof_handler.active_cell_iterators())
1200  {
1201  for (const auto &f : dof_handler.active_cell_iterators())
1202  {
1203  SauterQuadRule(system_matrix,
1204  this->dlp,
1205  bem_values,
1206  e,
1207  f,
1208  this->mapping,
1209  this->mapping);
1210 
1211  SauterQuadRule(system_rhs_matrix,
1212  this->slp,
1213  bem_values,
1214  e,
1215  f,
1216  this->mapping,
1217  this->mapping);
1218 
1219  ++pd;
1220  }
1221  }
1222 
1223  // Calculate the right-hand side vector.
1224  system_rhs_matrix.vmult(system_rhs, neumann_bc);
1225  }
